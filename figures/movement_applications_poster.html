movement applications (poster)

A poster figure with 7 panels—split across 4 rows—showcasing movement application examples.

Rows 1–3 cover the elevated plus maze (EPM) dataset, whereas row 4 covers the rotating-mouse eye-tracking dataset:

  1. a raw-vs-clean position trace (A)
  2. centroid trajectory and occupancy heatmap side-by-side (B, C)
  3. ROI time-fractions overlaid on the maze frame alongside a head-angle polar histogram (D, E).
  4. the four tracked keypoints (plus the pupil centroid) on an example video frame (F), and the pupil centroid’s horizontal velocity over time (G).

Load and prepare data

EPM dataset

The EPM dataset ships with movement’s sample data. We apply a three-step cleaning pipeline (confidence filter, rolling median, linear gap interpolation) and keep both the raw and clean positions so we can compare them in the first panel.

We also compute the time fraction spent in each ROI and the head direction angle for the clean positions, which will be visualized in the third figure row.

Code 
# Dataset and background frame.
ds = sample_data.fetch_dataset(
    "DLC_single-mouse_EPM.predictions.h5", with_video=True
)
frame = plt.imread(ds.frame_path)  # 1300 x 1028 px maze image for heatmap/ROI panels
frame_h, frame_w = frame.shape[:2]

# Video frame at the end of trajectory time window
# (to use as background for the trajectory panel).
video = sio.load_video(ds.video_path)
traj_end_sec = 250
traj_frame = video[int(traj_end_sec * ds.fps)]

# Cleaning pipeline.
pos_conf = filter_by_confidence(ds.position, ds.confidence, threshold=0.99)
pos_smooth = rolling_filter(pos_conf, window=9, statistic="median", min_periods=3)
pos_clean = interpolate_over_time(pos_smooth, method="linear", max_gap=5)

# Stack raw and clean for the trace panel.
positions = xr.concat([ds.position, pos_clean], dim="data")
positions.coords["data"] = ["raw", "clean"]

# Regions of interest.
rois = load_rois(data_dir() / "EPM_rois.geojson")

# Time fraction per ROI (mouse considered "in" a region only when both
# the centre and tailbase keypoints are inside it simultaneously).
roi_occ = compute_region_occupancy(pos_clean, rois)
mouse_in_roi = roi_occ.sel(
    keypoint=["centre", "tailbase"],
    individual="individual_0",
).all(dim="keypoint")

pct_time_in_roi = (
    mouse_in_roi.sum(dim="time") / mouse_in_roi.sizes["time"] * 100
)

# Head direction angle (radians, clockwise from positive x-axis).
head_angle = compute_forward_vector_angle(
    pos_clean.sel(individual="individual_0"),
    left_keypoint="left_ear",
    right_keypoint="right_ear",
    reference_vector=(1, 0),
    camera_view="top_down",
    in_degrees=False,
)

Eye-tracking dataset

We use a dataset that’s already part of movement’s sample data and has been thoroughly explored in movement’s ‘Pupil tracking’ example. Here we extract the first video frame with tracked keypoints and the pupil centroid, as well as the pupil centroid’s horizontal velocity over a specific window (normalised by the mean eye position to remove head/camera motion).

Code 
# Load the black (dark surround) eye-tracking dataset together with its video.
eye_ds = sample_data.fetch_dataset(
    "DLC_rotating-mouse_eye-tracking_stim-black.predictions.h5",
    with_video=True,
)
frame0 = sio.load_video(eye_ds.video_path)[0]

# Panel F data: keypoints at the first frame and the pupil centroid.
pos0 = eye_ds.position.isel(time=0).squeeze()
centroid0 = pos0.sel(keypoint=["pupil-L", "pupil-R"]).mean("keypoint")

# Panel G data: x-velocity of the normalized pupil centroid over time.
eye_mid = eye_ds.position.sel(keypoint=["eye-L", "eye-R"]).mean("keypoint")
pos_norm = eye_ds.position - eye_mid
pupil_c = pos_norm.sel(keypoint=["pupil-L", "pupil-R"]).mean("keypoint")
velocity = compute_velocity(pupil_c)
vx = velocity.sel(space="x").squeeze().sel(time=slice(10, 25))

Build the figure

Code 
fig = plt.figure(figsize=figure_size("double", height=17.5))
gs = fig.add_gridspec(
    4, 2,
    height_ratios=[2, 5, 5, 3.5],
    hspace=0.5,
    wspace=0.3,
    left=0.1, right=0.95, top=0.95, bottom=0.05,
)

ax_trace = fig.add_subplot(gs[0, :])           # row 0 -- full width
ax_traj  = fig.add_subplot(gs[1, 0])           # row 1 left
ax_heat  = fig.add_subplot(gs[1, 1])           # row 1 right
ax_roi   = fig.add_subplot(gs[2, 0])           # row 2 left
ax_polar = fig.add_subplot(gs[2, 1], projection="polar")  # row 2 right

# Row 3 keeps the pupil pair's original narrow-frame / wide-trace proportion.
gs_pupil = gs[3, :].subgridspec(1, 2, width_ratios=[5, 11], wspace=0.25)
ax_frame = fig.add_subplot(gs_pupil[0, 0])     # row 3 left
ax_vel   = fig.add_subplot(gs_pupil[0, 1])     # row 3 right

# ------------------------------------------------------------------ #
# Panel A: raw vs clean snout y-position                              #
# ------------------------------------------------------------------ #
pos_sel = positions.sel(
    keypoint="snout",
    individual="individual_0",
    space="y",
    time=slice(240, 300),
)
pos_sel.sel(data="raw").plot.line(
    ax=ax_trace, color=get_color("orange", "Pastel2"), alpha=1.0,
    label="Raw snout position",
)
pos_sel.sel(data="clean").plot.line(
    ax=ax_trace, color=get_color("teal", "Set2"), alpha=1.0,
    label="Clean (after dropping outliers and smoothing)",
)
ax_trace.set_xlim(float(pos_sel.time.min()), float(pos_sel.time.max()))
ax_trace.set_title("")
ax_trace.set_xlabel("time (s)")
ax_trace.set_ylabel("y position (px)")
# Anchor the legend on top of the panel (where the title would normally go) so it doesn't overlap the data.
ax_trace.legend(
    loc="lower center", bbox_to_anchor=(0.5, 1.0), frameon=True, framealpha=0.6, ncols=2
)

# ------------------------------------------------------------------ #
# Panel B: centroid trajectory                                        #
# ------------------------------------------------------------------ #
ax_traj.imshow(traj_frame.mean(axis=2), cmap="gray")
plot_centroid_trajectory(
    pos_clean.sel(time=slice(traj_end_sec - 10, traj_end_sec)),
    individual="individual_0",
    keypoints=["centre", "tailbase", "left_ear", "right_ear"],
    ax=ax_traj,
    alpha=0.8,
    cmap="viridis",
)
# anchor="W" left-justifies the equal-aspect image within its divider cell so
# its left edge lines up with panel A's left plot-area edge.
move_colorbar_to_divider(ax_traj, "time (s)", anchor="W")
ax_traj.set_xlim(0, frame_w)
ax_traj.set_ylim(frame_h, 0)
ax_traj.set_title("Centroid trajectory")
ax_traj.axis("off")

# ------------------------------------------------------------------ #
# Panel C: occupancy heatmap                                          #
# ------------------------------------------------------------------ #

# Occupancy histogram bins (30 px cells covering the full frame).
BIN_PIX = 30
bins = [
    np.arange(0, frame_w + BIN_PIX, BIN_PIX),
    np.arange(0, frame_h + BIN_PIX, BIN_PIX),
]

ax_heat.imshow(frame.mean(axis=2), cmap="gray")
_, ax_heat, _ = plot_occupancy(
    da=pos_clean,
    ax=ax_heat,
    alpha=0.8,
    bins=bins,
    cmin=10,
    norm="log",
    cmap="turbo",
)
move_colorbar_to_divider(ax_heat, "# frames")
ax_heat.set_xlim(0, frame_w)
ax_heat.set_ylim(frame_h, 0)
ax_heat.set_title("Occupancy heatmap")
ax_heat.axis("off")

# ------------------------------------------------------------------ #
# Panel D: time in ROIs                                               #
# ------------------------------------------------------------------ #
roi_colors = [
    get_color("teal", "Dark2") if roi.name.startswith("open")
    else get_color("pink", "Dark2") for roi in rois
]

ax_roi.imshow(frame.mean(axis=2), cmap="gray")
for i, roi in enumerate(rois):
    roi.plot(
        ax=ax_roi,
        edgecolor=roi_colors[i],
        facecolor="none",
    )
    cx, cy = roi.region.centroid.x, roi.region.centroid.y
    ax_roi.text(
        cx, cy,
        f"{float(pct_time_in_roi[i]):.0f}%",
        color=roi_colors[i],
        ha="center",
        va="center",
        fontsize="small",
    )
ax_roi.set_xlim(0, frame_w)
ax_roi.set_ylim(frame_h, 0)
ax_roi.set_title("Time spent in ROIs")
ax_roi.axis("off")

# Reserve an invisible colorbar-sized strip so this panel's image box matches
# the colorbar-bearing trajectory panel directly above it. anchor="W" keeps the
# image left-justified, matching panel A's left edge (as in the trajectory panel).
reserve_colorbar_space(ax_roi, anchor="W")

# ------------------------------------------------------------------ #
# Panel E: head direction polar histogram                             #
# ------------------------------------------------------------------ #
BIN_WIDTH_DEG = 15
n_bins = int(360 / BIN_WIDTH_DEG)

head_angle.sel(time=slice(150, 300)).plot.hist(
    bins=np.linspace(-np.pi, np.pi, n_bins + 1),
    ax=ax_polar,
    density=True,
    color=get_color("teal", "Set2"),
    edgecolor="white",
    linewidth=1.0,
)
ax_polar.set_theta_direction(-1)   # clockwise
ax_polar.set_theta_offset(0)       # 0 deg at the right
ax_polar.set_xlabel("")
ax_polar.set_yticklabels([])  # auto radial gridlines, hide density labels

tick_angles = np.linspace(0, 2 * np.pi, 9)[:-1]
tick_labels_deg = list(range(0, 181, 45)) + list(range(0, -180, -45))[-1:0:-1]
ax_polar.set_xticks(tick_angles)
ax_polar.set_xticklabels([f"{t}\N{DEGREE SIGN}" for t in tick_labels_deg])
ax_polar.set_title("Head direction", pad=45)

# ------------------------------------------------------------------ #
# Panel F: example eye frame with tracked keypoints + centroid        #
# ------------------------------------------------------------------ #
ax_frame.imshow(frame0, cmap="gray")
# All 4 tracked keypoints share one color and one label -> a single legend entry.
ax_frame.scatter(
    pos0.sel(space="x"),
    pos0.sel(space="y"),
    s=120,
    color=get_color("teal", "Set2"),
)
ax_frame.scatter(
    centroid0.sel(space="x"),
    centroid0.sel(space="y"),
    s=120,
    color=get_color("orange", "Set2"),
)
ax_frame.invert_yaxis()  # the dataset was captured flipped

# Hide ticks and spines but keep the axis labels
ax_frame.set_xticks([])
ax_frame.set_yticks([])
for spine in ax_frame.spines.values():
    spine.set_visible(False)

ax_frame.set_xlabel("x")
ax_frame.set_ylabel("y")
ax_frame.set_anchor("W")  # left-justify to align with panel A's left edge

# ------------------------------------------------------------------ #
# Panel G: pupil centroid x-velocity over time                        #
# ------------------------------------------------------------------ #
vx.plot.line(
    ax=ax_vel, color=get_color("orange", "Set2"), label="Pupil centroid x velocity"
)
ax_vel.set_xlim(float(vx.time.min()), float(vx.time.max()))
ax_vel.set_title("Pupil centroid x velocity")
ax_vel.set_xlabel("time (s)")
ax_vel.set_ylabel("velocity (px/s)");
